#version 440

out vec4 position;
out vec2 texco;
out vec3 norm;

uniform mat4 transform;
uniform mat3 normal_transform;

uniform vec3 angle;

in vec4 pos;
in vec3 normal;
in vec2 texCoord0;

void main()
{
	norm = normal_transform * normal;
	texco = texCoord0;
	vec4 p = pos + vec4(sin(angle * vec3(10.0, 10.0, 0.1)) * 0.1 * texCoord0.y, 0.0);
	position = transform * p;
	gl_Position = transform * p;
}

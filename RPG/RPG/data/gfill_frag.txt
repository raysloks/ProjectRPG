#version 130

varying vec2 texco;
varying vec3 norm;

uniform sampler2D tex;

uniform vec3 light;

void main()
{
	float intensity = dot(light, normalize(norm));
	intensity = max(0.0, intensity);

	intensity += 0.1;

	vec3 color = texture(tex, texco).rgb;

	color *= intensity;

	gl_FragData[0] = vec4(color, 1.0);
}
